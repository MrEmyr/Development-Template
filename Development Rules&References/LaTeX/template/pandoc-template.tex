% Pandoc LaTeX template for repo documentation (Phase 4)
% This is a Pandoc template (uses Pandoc template variables like $title$), not a standalone LaTeX doc.
% Macros are included via --include-in-header=.../macros.tex in the build script.

\documentclass[11pt]{article}

\usepackage[margin=1in]{geometry}
\usepackage[T1]{fontenc}
\usepackage{lmodern}
\usepackage{xcolor}
\usepackage{hyperref}
\usepackage{listings}

\providecommand{\tightlist}{}

\hypersetup{
  colorlinks=true,
  linkcolor=black,
  urlcolor=black
}

$if(title)$
\title{$title$}
$endif$
$if(author)$
\author{$for(author)$$author$$sep$ \and $endfor$}
$endif$
$if(date)$
\date{$date$}
$else$
\date{}
$endif$

\begin{document}
$if(title)$
\maketitle
$endif$

$if(toc)$
\tableofcontents
\newpage
$endif$

$body$
\end{document}
